#Quest Maintenance
character_event = {
	id = clash_of_kings.5099

	is_triggered_only = yes
	hide_window = yes
	
	trigger = { 
		character = 77041 
		has_character_flag = danys_eastern_quest
		has_game_rule = { name = shattered_realms value = off }
	}
	
	immediate = { 
		if = {
			limit = { had_character_flag = { flag = dany_in_qarth days = 130 } }
			character_event = { id = clash_of_kings.5200 }
			break = yes
		}
		if = {
			limit = { 
				has_character_flag = emancipation_of_yunkai
				war = no
			}
			if = {
				limit = { k_yunkai = { has_law = slavery_2 } }
				k_yunkai = {
					reverse_unsafe_war = {
						target = ROOT
						casus_belli = war_of_emancipation
					}
				}
				if = {
					limit = { war = no }
					narrative_event = { id = clash_of_kings.57 }
				}
				break = yes
			}
			else = {
				narrative_event = { id = clash_of_kings.57 }
				break = yes
			}
		}
		if = {
			limit = { 
				has_character_flag = liberation_of_meereen
				war = no
			}
			k_meereen = {
				if = {
					limit = { 
						NOT = { has_law = slavery_2 } 
						NOT = { has_game_rule = { name = slavery value = off } }
					}
					holder_scope = { set_character_flag = passive_slave_law_change }
					add_law = slavery_2
				}
				reverse_unsafe_war = {
					target = ROOT
					casus_belli = war_of_liberation
				}
			}
			break = yes
		}	
		if = {
			limit = { 
				has_global_flag = enslavement_of_astapor
				k_yunkai = { holder_scope = { war = no } }
			}
			k_yunkai = { 
				holder_scope = {
					if = {
						limit = { NOT = { character = ROOT } }
						k_yunkai = { 
							if = {
								limit = { 
									NOT = { has_law = slavery_2 } 
									NOT = { has_game_rule = { name = slavery value = off } }
								}
								holder_scope = { set_character_flag = passive_slave_law_change }
								add_law = slavery_2
							}
						}	
						if = {
							limit = { NOT = { has_landed_title = k_astapor } }
							prestige = 200
							if = {
								limit = { NOT = { has_game_rule = { name = slavery value = off } } }
								character_event = { id = slavery.1000 tooltip = "EVTTOOLTIPSLAVERY1000" } # The attacker gains a slave camp level
							}
							k_astapor = {
								holder_scope = {
									prestige = -400	
								}
							}	
							k_astapor = {
								usurp_title_plus_barony_if_unlanded = { target = ROOT type = invasion }
								add_law = slavery_2
							}
						}
						ROOT = {
							set_character_flag = enslavement_of_meereen
							narrative_event = { id = feast_for_crows.53 days = 10 }
						}
						k_meereen = {								
							reverse_unsafe_war = {
								target = PREV
								casus_belli = war_of_enslavement
							}
						}
						clr_global_flag = enslavement_of_astapor
						any_war = {
							if = {
								limit = { 
									OR = {
										AND = {
											attacker = { character = ROOT }
											using_cb = slave_raid 
										}	
										defender = { character = ROOT } 
									}	
								}	
								end_war = invalid
							}	
						}
					}
					if = {
						limit = { character = ROOT }
						clr_global_flag = enslavement_of_astapor
						set_character_flag = dany_liberation_quest
						narrative_event = { id = feast_for_crows.57 } 
					}	
				}
			}			
			break = yes
		}
		if = {
			limit = { 
				has_character_flag = enslavement_of_meereen
				war = no
			}
			if = {
				limit = { 
					k_yunkai = { has_law = slavery_2 } 
					NOT = { has_game_rule = { name = slavery value = off } }
				}
				k_yunkai = { 
					holder_scope = {
						k_meereen = {								
							reverse_unsafe_war = {
								target = PREV
								casus_belli = war_of_enslavement
							}
						}
					}
				}
				if = {
					limit = { war = no }
					set_character_flag = dany_liberation_quest
					narrative_event = { id = feast_for_crows.57 } 
				}
				break = yes
			}
			else = {
				set_character_flag = dany_liberation_quest
				narrative_event = { id = feast_for_crows.57 } 
				break = yes
			}
		}	
	} 
	
	option = {
		name = OK
	}
	
	after = {
		character_event = { id = clash_of_kings.5099 days = 50 } #check every 50 days
	}
}
