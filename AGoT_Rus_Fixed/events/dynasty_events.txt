	is_triggered_only = yes
	only_independent = yes

		higher_tier_than = DUKE
		government = feudal_government
		OR = {
			tier = EMPEROR
			primary_title = { NOT = { check_variable = { which = "de_facto_empire" value = 1 } } }
		}
		primary_title = {
			OR = {
				has_law = succ_primogeniture
				has_law = succ_gavelkind
			}	
			has_law = true_cognatic_succession
		}
		current_heir = {
			is_ruler = yes
			is_female = yes
			is_child_of = ROOT
			has_character_modifier = nominated_heir
			dynasty = ROOT
		}
		NOT = { culture_group = dornish }
		NOT = { has_dynasty_flag = prefers_cognatic }
	}
	
	immediate = {	
		current_heir = { save_event_target_as = female_heir }
		#First check if heir under agnatic is a close relative and then flag
		any_demesne_title = { 
			limit = { tier = ROOT }
			gender_succ = agnatic 
		}
		recalc_succession = yes
		current_heir = {
			if = {
				limit = {
					is_female = no
					is_close_relative = ROOT
					NOT = { is_child_of = event_target:female_heir }
					NOT = { is_grandchild_of = event_target:female_heir }
				}
				#set_character_flag = male_pretender
				ROOT = { add_trait = dead_target }
			}
		}
		#If no male heir is found then end event
		if = { 
			limit = { NOT = { trait = dead_target } }
			any_demesne_title = { limit = { tier = ROOT } gender_succ = true_cognatic }
			recalc_succession = yes	
			current_heir = { remove_character_modifier = nominated_heir }	
			break = yes
		}
		#If a viable male heir is found then select primary coup actor
		remove_trait = dead_target
		current_heir = { #Check the pretender
			if = {
				limit = {
					liege = { character = ROOT }					
					prisoner = no
					is_incapable = no
					in_command = no
					OR = {
						AND = { #Is a powerful lord in capital
							OR = {
								diplomacy = 10
								intrigue = 10
							}
							OR = {
								higher_tier_than = BARON
								prestige = 500
								trait = dragon_rider
							}	
							location = { ROOT = { capital_scope = { province_id = PREVPREV } } }
						}
						liege = { regent = { character = PREVPREV } }
						is_councillor = yes
						is_voter = yes
					}
					NOT = { trait = content }
					OR = {
						trait = ambitious
						trait = ruthless
						trait = just
						NOT = { opinion = { who = event_target:female_heir value = 25 } }
					}
					NAND = {
						trait = honorable
						NOT = { trait = just }
						is_close_relative = ROOT
					}	
					NOT = { is_married = event_target:female_heir }
				}
				if = {
					limit = { ai = no } #record players current titles in case they refuse
					any_demesne_title = { set_title_flag = my_previous_title }
				}
				character_event = { id = dynasty.4 days = 1 }
				break = yes
			}
		}
		if = {
			limit = { 
				current_heir = {
					OR = {
						ai = yes 
						is_adult = no
						is_incapable = yes
						prisoner = yes
					}						
					NAND = { #Claimants who like their liege or are honorable can't be used if they are strong
						is_adult = yes
						is_incapable = no
						OR = {
							higher_tier_than = COUNT
							trait = authoritative
							martial = 15
						}					
						OR = {
							trait = content
							AND = {
								trait = honorable
								NOT = { trait = just }
								is_close_relative = ROOT
							}	
							AND = {
								NOT = { trait = ambitious }
								NOT = { trait = ruthless }
								NOT = { trait = just }
								opinion = { who = event_target:female_heir value = 25 }
							}	
						}
					}
				} 
			}
			#Then check hand/castellan
			job_castellan = {
				if = {
					limit = {					
						in_command = no
						NOT = { is_grandparent_of = event_target:female_heir }
						NOT = { is_parent_of = event_target:female_heir }
						NAND = {
							trait = honorable
							NOT = { trait = just }
						}
						NOT = { trait = content }
						OR = { #Motives
							trait = just	#Dislikes law breaking
							AND = { #Likes male heir and either dislikes old king or female heir
								ROOT = { current_heir = { reverse_opinion = { who = ROOT value = 0 } } }

							NOT = { is_parent_of = event_target:female_heir }
