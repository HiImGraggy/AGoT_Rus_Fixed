character_event = {
	id = 73230
	desc = "EVTDESC73230"	#I want to be lord_spiritual, I am better...
	picture = GFX_evt_spymaster

	only_men = yes
	capable_only = yes
	prisoner = no
	min_age = 16
	
	trigger = {
		NOT = { liege = { is_married = ROOT } }
		independent = no
		NOT = { demesne_size = 1 }
		is_councillor = no
		NOT = { is_voter = yes }
		trait = maester
		#NOT = { liege = { has_character_flag = no_council_requests } }
		NOT = { liege = { has_character_flag = asked_lord_spiritual_petition } }
		NOT = { liege = { has_landed_title = e_iron_throne } }
		NOT = { has_character_flag = asked_for_job }
		liege = {
			job_physician = {
				same_liege = ROOT
				NOT = { learning = ROOT }
				NOT = { liege = { regent = { character = PREVPREV } } }
			}
		}	
		liege = { religion = ROOT }
		NOT = {
			has_opinion_modifier = { who = ROOT modifier = opinion_fired_from_council }
		}
		can_hold_title = job_physician
	}
	
	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 0.25
			trait = ambitious
		}
	}
	
	immediate = {
		set_character_flag = asked_for_job
		liege = {  set_character_flag = asked_lord_spiritual_petition }
	}
	
	option = {
		name = "EVTOPTA73230" 
		liege = { letter_event = { id = 73231 tooltip = EVTTOOLTIP73231 } }
	}
}
